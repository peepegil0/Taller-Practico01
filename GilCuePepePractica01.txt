    for $libro in /biblioteca/libro[@categoria='programacion' and precio > 30]/titulo order by $libro descending

    <resultado>

    </resultado>

    for $autor in distinct-values (//autor)

3. Generar una estructura de lista HTML (<ul>) donde cada elemento sea un libro. Si el libro es posterior a 2022, debe incluir el texto "¡Novedad!" al lado del título. El precio debe aparecer con el símbolo €. -->

    for $libro in //libro 
    let $titulo  := $libro/titulo/text()
    let $precio := number($libro/precio)
    let anio   := number($libro/publicacion)
    return 
    <resultado>
        <ul>
            <li>
                {$titulo} {if ($anio > 2022) then "¡Novedad!" else ""} - {$precio}
            </li>
        </ul>
    </resultado>
